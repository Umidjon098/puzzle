üéµ Ovoz Effektlari

HTML5 Audio API yordamida ovoz effektlarini qo'shamiz.

JavaScript - Ovoz Menejeri 
/**
 * Ovozlarni boshqaruvchi klass
 */
class SoundManager {
    constructor() {
        this.sounds = {};
        this.enabled = true;
    }
    
    /**
     * Ovozni yuklash
     */
    load(name, url) {
        const audio = new Audio(url);
        audio.preload = 'auto';
        this.sounds[name] = audio;
    }
    
    /**
     * Ovozni ijro etish
     */
    play(name, volume = 1.0) {
        if (!this.enabled || !this.sounds[name]) return;
        
        const sound = this.sounds[name].cloneNode();
        sound.volume = volume;
        sound.play().catch(err => {
            console.log('Ovoz ijro bo\'lmadi:', err);
        });
    }
    
    /**
     * Ovozni o'chirish/yoqish
     */
    toggle() {
        this.enabled = !this.enabled;
        return this.enabled;
    }
}

// Global sound manager
const soundManager = new SoundManager();

// Ovozlarni yuklash

soundManager.load('win', 'assets/sounds/win.wav');

üé¨ Katak Animatsiyasi

Kataklarni almashtirishda silliq animatsiya qo'shamiz.

JavaScript - Silliq Almashtirish 
/**
 * Kataklarni animatsiya bilan almashtirish
 */
function swapTilesWithAnimation(tile1, tile2) {
    // Click ovozini ijro etish
    soundManager.play('click', 0.3);
    
    // Animatsiya klassini qo'shish
    tile1.classList.add('swapping');
    tile2.classList.add('swapping');
    
    // Kichik kechikish
    setTimeout(() => {
        // Background pozitsiyasini almashtirish
        const tempBg = tile1.style.backgroundPosition;
        tile1.style.backgroundPosition = tile2.style.backgroundPosition;
        tile2.style.backgroundPosition = tempBg;
        
        // Data atributlarini almashtirish
        const tempPos = tile1.dataset.position;
        tile1.dataset.position = tile2.dataset.position;
        tile2.dataset.position = tempPos;
        
        // Animatsiya klassini olib tashlash
        setTimeout(() => {
            tile1.classList.remove('swapping');
            tile2.classList.remove('swapping');
        }, 300);
    }, 150);
}

üåü Zarrachalar Effekti

G'alaba qilganda zarrachalar effektini qo'shamiz (konfetti).

JavaScript - Konfetti Effekti 
/**
 * Konfetti zarrachalarini yaratadi
 */
function createConfetti() {
    const colors = ['#6366f1', '#8b5cf6', '#10b981', '#f59e0b', '#ef4444'];
    const confettiCount = 50;
    
    for (let i = 0; i < confettiCount; i++) {
        const confetti = document.createElement('div');
        confetti.className = 'confetti';
        
        // Tasodifiy rang
        const color = colors[Math.floor(Math.random() * colors.length)];
        confetti.style.backgroundColor = color;
        
        // Tasodifiy pozitsiya
        confetti.style.left = Math.random() * 100 + '%';
        confetti.style.animationDelay = Math.random() * 3 + 's';
        confetti.style.animationDuration = (Math.random() * 3 + 2) + 's';
        
        document.body.appendChild(confetti);
        
        // 5 soniyadan keyin olib tashlash
        setTimeout(() => confetti.remove(), 5000);
    }
}

// CSS
/*

class MultimediaManager {
    constructor() {
        this.soundManager = new SoundManager();
        this.initSounds();
    }
    
    initSounds() {
        this.soundManager.load('click', 'assets/sounds/click.wav');
        this.soundManager.load('win', 'assets/sounds/win.wav');
        this.soundManager.load('shuffle', 'assets/sounds/shuffle.wav');

        this.soundManager.play('click', 0.3);

            setTimeout(() => confetti.remove(), 5000);
        }
    }

        const message = document.createElement('div');
